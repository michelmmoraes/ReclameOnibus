66           mBuilder.setContentIntent(resultPendingIntent); 
67           mNotificationManager.notify(idNotificacao, mBuilder.build());

195      private RespostaCenario detectarCenarioAntigo(Cidadao cidadao) { 
196          RespostaCenario respostaCenario = new RespostaCenario(); 
197          //publishProgress("buscando paradas..."); 
198          RespostaParada respostaParada = detectarParadaAntigo(cidadao); 
199          //publishProgress("paradas ok"); 
200          if (respostaParada.isEstaNaParada()) { 
201              // O usuário está em uma parada 
202              respostaCenario.setIdCenario(RespostaCenario.PARADA); 
203              respostaCenario.setParada(respostaParada.getParada()); 
204              return respostaCenario; 
205          } else { 
206              notificar(11111111, "Parada=false", 
207                      "Você não está em uma parada! Testar abrir TelaParada", "parada :(", TelaParada.class); 
208              //publishProgress("buscando ônibus..."); 
209              RespostaOnibus respostaOnibus = detectarOnibusAntigo(cidadao); 
210              //publishProgress("ônibus ok"); 
211              if (respostaOnibus.isEstaNoOnibus()) { 
212                  // O usuário está em um ônibus 
213                  respostaCenario.setIdCenario(RespostaCenario.ONIBUS); 
214                  respostaCenario.setOnibus(respostaOnibus.getOnibus()); 
215                  return respostaCenario; 
216              } else { 
217                  // O usuário está em qualquer outro lugar 
218                  respostaCenario.setIdCenario(RespostaCenario.OUTRO); 
219                  return respostaCenario; 
220              } 
221          }

224      private RespostaParada detectarParadaAntigo(Cidadao cidadao) { 
225          HttpGet httpGet = new HttpGet("http://api.olhovivo.sptrans.com.br/v0" + 
226                  "/Parada/Buscar?termosBusca="); 
227          Log.e("parada", "detectando todas as paradas..."); 
228          try { 
229              HttpResponse httpResponse = httpClient.execute(httpGet); 
230              Log.e("parada", "HttpResponse httpResponse = httpClient.execute(httpGet);"); 
231              HttpEntity httpEntity = httpResponse.getEntity(); 
232              Log.e("parada", "HttpEntity httpEntity = httpResponse.getEntity();"); 
233              JSONArray jsonArray = new JSONArray(EntityUtils.toString(httpEntity)); 
234              Log.e("parada", "JSONArray jsonArray = new JSONArray(EntityUtils.toString(httpEntity));"); 
235              //publishProgress("qtd paradas=" + jsonArray.length()); 
236              Log.e("parada", "foram encontradas " + jsonArray.length() + " paradas."); 
237              for (int i = 0; i < jsonArray.length(); i++) { 
238                  Log.e("parada", "parada encontrada nº" + i); 
239                  if (cidadao.getLocalizacao().getLatitude() == 
240                          jsonArray.getJSONObject(i).getDouble("Latitude")) { 
241                      if (cidadao.getLocalizacao().getLongitude() == 
242                              jsonArray.getJSONObject(i).getDouble("Longitude")) { 
243                          // O cidadão está em uma parada 
244                          RespostaParada respostaParada = new RespostaParada(); 
245                          respostaParada.setEstaNaParada(true); 
246                          Parada parada = new Parada(); 
247                          parada.setCodigoParada(jsonArray.getJSONObject(i).getInt("CodigoParada")); 
248                          parada.setNome(jsonArray.getJSONObject(i).getString("Nome")); 
249                          parada.setEndereco(jsonArray.getJSONObject(i).getString("Endereco")); 
250                          Localizacao localizacao = new Localizacao(); 
251                          localizacao.setLatitude(jsonArray.getJSONObject(i).getDouble("Latitude")); 
252                          localizacao.setLongitude(jsonArray.getJSONObject(i).getDouble("Longitude")); 
253                          parada.setLocalizacao(localizacao); 
254                          respostaParada.setParada(parada); 
255                          Log.e("parada", "parada encontrada!ok!"); 
256                          return respostaParada; 
257                      } 
258                  } 
259              } 
260          } catch (Exception e) { 
261              Log.e("parada", "ERRO: " + e.getMessage()); 
262              e.printStackTrace(); 
263          } 
264          RespostaParada respostaParada = new RespostaParada(); 
265          respostaParada.setEstaNaParada(false); 
266          Log.e("parada", "busca de paradas foi concluída."); 
267          return respostaParada; 
268      }

344      private List<Linha> buscarTodasLinhasAntigo() { 
345          List<Linha> listaLinhas = new ArrayList<>(); 
346          JSONArray arrayTodasLinhas = new JSONArray(); 
347          Log.e("linha", "buscando linhas..."); 
348          for (int i = 0; i < 10; i++) { 
349              HttpGet httpGet = new HttpGet("http://api.olhovivo.sptrans.com.br/v0" + 
350                      "/Linha/Buscar?termosBusca=" + i); 
351              Log.e("linha", "buscando linhas... letreiros começando em " + i); 
352              try { 
353                  HttpResponse httpResponse = httpClient.execute(httpGet); 
354                  HttpEntity httpEntity = httpResponse.getEntity(); 
355                  String respostaJSON = EntityUtils.toString(httpEntity); 
356                  JSONArray jsonArray = new JSONArray(respostaJSON); 
357                  arrayTodasLinhas.put(jsonArray); 
358                  Log.e("linha", "letreiros começando em " + i + " têm " + jsonArray.length() + " linhas."); 
359                  for (int j = 0; j < jsonArray.length(); j++) { 
360                      JSONObject linhaDaVez = jsonArray.getJSONObject(j); 
361                      Log.e("linha", "letreiro começa com " + i + ", linha nº " + j 
362                              + ", codigoLinha=" + linhaDaVez.getInt("CodigoLinha")); 
363                      Linha linha = new Linha(); 
364                      linha.setCodigoLinha(linhaDaVez.getInt("CodigoLinha")); 
365                      linha.setCircular(linhaDaVez.getBoolean("Circular")); 
366                      linha.setLetreiro(linhaDaVez.getString("Letreiro")); 
367                      linha.setSentido(linhaDaVez.getString("Sentido")); 
368                      linha.setTipo(linhaDaVez.getString("Tipo")); 
369                      linha.setDenominacaoTPTS(linhaDaVez.getString("DenominacaoTPTS")); 
370                      linha.setDenominacaoTSTP(linhaDaVez.getString("DenominacaoTSTP")); 
371                      listaLinhas.add(linha); 
372                  } 
373              } catch (IOException | JSONException e) { 
374                  publishProgress("ERRO AO BUSCAR LINHAS!\n" + e.getMessage()); 
375                  Log.e("linha", "buscando linhas... erro:" + e.getMessage()); 
376                  e.printStackTrace(); 
377              } 
378          } 
379          Log.e("linha", "foram encontradas " + listaLinhas.size() + " linhas."); 
380          return listaLinhas; 
381      }
